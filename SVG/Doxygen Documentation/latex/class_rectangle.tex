\doxysubsection{Detailed Description}

\begin{DoxyParams}{Parameters}
{\em x} & The x property describes the horizontal coordinate of the position of the element. \\
\hline
{\em y} & The y property describes the vertical coordinate of the position of the element. \\
\hline
{\em width} & This property specifies the width of the rectangle. Negative values for \textquotesingle{}width\textquotesingle{} are illegal. \\
\hline
{\em height} & This property specifies the height of the rectangle. Negative values for \textquotesingle{}height\textquotesingle{} are illegal. \\
\hline
{\em rx} & The rx property describes the horizontal curve radius of the rectangle. A negative value for rx is illegal. \\
\hline
{\em ry} & The ry property describes the vertical curve radius of the rectangle. A negative value for ry is illegal. \\

\end{DoxyParams}
